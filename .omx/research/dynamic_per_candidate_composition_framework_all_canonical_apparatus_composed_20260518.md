```python
def compose_optimal_per_candidate(
    candidate_archive_sha256: str,
    candidate_lane_id: str,
    asymptotic_approach: AsymptoticApproach,  # PLATEAU_ADJACENT / FRONTIER_PURSUIT / ASYMPTOTIC_PURSUIT per Catalog #309
    budget_usd: float = 1.0,
    repo_root: Path | None = None,
) -> CompositionPlan:
    """
    Compose the optimal per-candidate plan from ALL canonical apparatus.

    Returns a typed CompositionPlan with:
    - per-byte-position primitive selection (which of the 14 primitives apply where)
    - per-primitive composition_alpha (sub-additive default per Catalog #322)
    - predicted ΔS band per primitive AND aggregate (Dykstra-feasible per Catalog #296)
    - cost estimate (Modal/Lightning/Vast.ai + CPU-only probe budget)
    - per-primitive legal-receiver-path classification per Catalog #6/L4
    - per-primitive HARD-EARNED vs CARGO-CULTED classification per Catalog #303
    - 6-hook wire-in declaration per Catalog #125
    - reactivation criteria per CLAUDE.md "Forbidden premature KILL"
    """
    # STEP 1: Load master_gradient anchor for candidate (or call extractor if missing)
    mg_anchor = load_master_gradient_for_archive(
        archive_sha256=candidate_archive_sha256,
        repo_root=repo_root,
    )
    if mg_anchor is None:
        # Per CLAUDE.md "Subagent coherence-by-default" + Catalog #245:
        # call tools/extract_master_gradient.py to register fp64 per-pair anchor;
        # cost ~$0 if CPU-only smoke; ~$0.30 if Modal A10G
        mg_anchor = extract_master_gradient_for_candidate(
            candidate_archive_sha256, candidate_lane_id, budget_usd
        )

    # STEP 2: Classify byte positions via 3-set or 6-set Venn per Catalog #319/#322
    # 3-set baseline: {HIGH_PAIR_INVARIANT, HIGH_PAIR_SPECIFIC, NEUTRAL}
    # 6-set extension per the N-set Venn design memo:
    # pair × region × class × frame × axis × component dimensions
    venn_classification = classify_archive_bytes_via_n_set_venn(
        archive_sha256=candidate_archive_sha256,
        master_gradient=mg_anchor,
        n=6,  # operator-tunable; default 6-set per N-set Venn design memo
        repo_root=repo_root,
    )

    # STEP 3: Identify hard-pair regions via sensitivity_map + xray observability
    hard_pair_atlas = build_hard_pair_sensitivity_atlas(
        archive_sha256=candidate_archive_sha256,
        master_gradient=mg_anchor,
        venn_classification=venn_classification,
        xray_layers=("rgb_input", "scorer_input", "scorer_output"),
    )

    # STEP 4: Run deterministic-optimizer bilevel (4-piece architecture)
    # per cross-stack synthesis §15 + the 4 sister design memos:
    # Phase 1 Fisher-precondition + Phase 2 Riemannian-Newton (smooth-pose)
    # + Phase 6 Tropical d_seg (boundary-class) + 3-set Venn (cell structure)
    bilevel_optimizer_state = run_deterministic_optimizer_4_piece(
        candidate_archive_sha256,
        master_gradient=mg_anchor,
        venn_classification=venn_classification,
        hard_pair_atlas=hard_pair_atlas,
        asymptotic_approach=asymptotic_approach,
    )

    # STEP 5: Per byte position, choose composition of ALL applicable primitives.
    # composition_alpha sub-additive default per Catalog #322
    # legal-receiver-path classification per Catalog #6 + L4 per Section 7 of this memo
    per_position_primitives: dict[ByteRange, list[CompositionPrimitive]] = {}
    for byte_range, venn_cell in venn_classification.items():
        applicable_primitives = enumerate_applicable_primitives_for_cell(
            venn_cell=venn_cell,
            asymptotic_approach=asymptotic_approach,
            budget_remaining_usd=budget_usd,
            hard_pair_atlas=hard_pair_atlas,
            bilevel_optimizer_state=bilevel_optimizer_state,
        )
        # Each primitive is one of:
        # - null_exploit (per src/tac/null_space_exploiter/)
        # - procedural_codebook (per src/tac/procedural_codebook_generator/)
        # - wyner_ziv (per src/tac/codec/wyner_ziv_layer.py + Catalog #319)
        # - freezing (per Catalog #779 compress-time SegNet/PoseNet freezing)
        # - A1-SPECIALIZED binary (per sister subagent acb1a03a8e4234022)
        # - hardware-codec (per per-axis matrix; CPU vs GPU)
        # - CPU/GPU asymmetry (per Hotz G1 cross-axis re-rank)
        # - stacking-with-orthogonal (per orthogonal-methods-audit #778)
        per_position_primitives[byte_range] = filter_by_legal_receiver_path(
            applicable_primitives,
            loc_budget=200,  # HNeRV parity L4 ≤200 LOC
            dep_budget=2,   # HNeRV parity L4 ≤2 deps
        )

    # STEP 6: Per cathedral autopilot rerank, apply 3-cascade decision tree
    # per Catalog #319 Q3 v2 (adjust_predicted_delta_for_venn_classification_v2):
    # Cascade 1 (PRIMARY): if Lagrangian-dual OptimalPerPairTreatmentPlan exists,
    #   REPLACE predicted_delta with planner's predicted_score_delta
    # Cascade 2 (DELIVERABILITY): else if DeliverabilityProof + Venn HIGH_PAIR_INVARIANT,
    #   apply per-tier byte-weighted reward (Tier 1 = 1.20x, Tier 2 = 1.10x,
    #   approved Tier 3 = 1.05x)
    # Cascade 3 (PASSTHROUGH): no plan + no proof => 1.0x passthrough
    cathedral_autopilot_verdict = apply_v2_cascade(
        candidate_archive_sha256,
        venn_classification=venn_classification,
        per_position_primitives=per_position_primitives,
        bilevel_optimizer_state=bilevel_optimizer_state,
    )

    # STEP 7: Emit CompositionPlan with:
    # - per-byte-position primitive selection + per-primitive composition_alpha
    # - aggregate predicted ΔS band (Dykstra-feasible per Catalog #296)
    # - cost estimate
    # - reactivation criteria per CLAUDE.md "Forbidden premature KILL"
    plan = build_composition_plan(
        candidate_archive_sha256=candidate_archive_sha256,
        candidate_lane_id=candidate_lane_id,
        asymptotic_approach=asymptotic_approach,
        budget_usd=budget_usd,
        per_position_primitives=per_position_primitives,
        cathedral_autopilot_verdict=cathedral_autopilot_verdict,
        bilevel_optimizer_state=bilevel_optimizer_state,
    )

    # STEP 8: Persist plan to canonical state ledger per Catalog #131/#138
    # (fcntl-locked JSONL append-only sister to Catalog #245 Modal call_id ledger)
    persist_composition_plan_to_ledger(plan, repo_root=repo_root)

    # STEP 9: Append council-deliberation anchor per Catalog #300 hook #5
    # (continual-learning posterior update)
    append_council_anchor_for_plan(plan, repo_root=repo_root)

    return plan
```
