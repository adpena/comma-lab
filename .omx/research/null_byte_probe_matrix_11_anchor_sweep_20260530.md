## 7. Sweep methodology (canonical Python script for reproducibility)

```python
from pathlib import Path
import json
import numpy as np

REPO_ROOT = Path("/Users/adpena/Projects/pact")
LEDGER = REPO_ROOT / ".omx/state/master_gradient_anchors.jsonl"

rows = []
with LEDGER.open("r") as fh:
    for ln in fh:
        ln = ln.strip()
        if not ln:
            continue
        rows.append(json.loads(ln))

# 11 anchors expected
assert len(rows) == 11

results = []
for idx, row in enumerate(rows, start=1):
    npy_path = Path(row["gradient_array_path"])
    if not npy_path.is_absolute():
        npy_path = REPO_ROOT / npy_path
    arr = np.load(npy_path, allow_pickle=False)
    # 2-D (N_bytes, 3) [seg, pose, rate] OR 1-D (N_bytes,) aggregate
    if arr.ndim == 2:
        # AGGREGATE NULL = byte where all 3 axes are exactly zero
        agg_null = int(np.sum(np.all(arr == 0.0, axis=1)))
    else:
        agg_null = int(np.sum(arr == 0.0))
    n_bytes = arr.shape[0]
    null_fraction = agg_null / n_bytes if n_bytes > 0 else 0.0
    results.append({
        "anchor_index": idx,
        "archive_sha_prefix": row["archive_sha256"][:10],
        "axis": row["measurement_axis"],
        "ledger_n_bytes": row["n_bytes"],
        "actual_n_bytes": n_bytes,
        "actual_agg_null_bytes": agg_null,
        "null_fraction": null_fraction,
        # CANONICAL DISCIPLINE: archive_zero_bytes ≠ master_gradient_null_bytes
        # (Slot QQ 2026-05-29 empirical anchor; pr106 16,909 gradient-null vs
        # 665 archive-zero). Per-anchor archive-zero MUST be measured separately
        # via byte-mutation smoke per Catalog #105 + #139.
    })

# All 11 anchors processed; gradient-null counts match Slot MM matrix EXACTLY:
# anchor 3 (a1): 16,037 / anchor 4 (pr101_lc_v2): 16,033 / anchor 5 (fec6): 16,292
# anchor 9 (pr106): 16,909 / anchor 10 (pr107): 15,987 / anchor 11 (fec6 CUDA): 16,292
```
